  onMomentumScrollEnd = e => {
    const offset = e.nativeEvent.contentOffset.x;
    const newIndex = this.rtlSafeIndex(Math.round(offset / this.state.width));
    if (newIndex === this.state.activeIndex) {
      return;
    }
    this.setState({ activeIndex: newIndex });
  };
 
  goToSlide = pageNum => {
